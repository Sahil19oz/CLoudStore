DROP DATABASE if exists BarclaysStore;
CREATE DATABASE BarclaysStore;
USE  BarclaysStore;

CREATE TABLE ADDRESS (
	ADDRESS_ID INT  NOT NULL auto_increment,
	LINE_1 VARCHAR (500) NOT NULL,
	LINE_2 VARCHAR (500),
	CITY VARCHAR(50) NOT NULL,
	STATE VARCHAR(50) NOT NULL,
	PIN VARCHAR(10) NOT NULL,
	LOC_ID INTEGER NOT NULL,
	constraint ADDRESS_ID_PK primary key ( ADDRESS_ID )
);

CREATE TABLE CUSTOMER (
	EMAIL_ID VARCHAR(50) ,
	NAME VARCHAR(50) NOT NULL,
	PASSWORD VARCHAR(70) NOT NULL,
	PHONE_NUMBER VARCHAR(10) NOT NULL UNIQUE,
	ADDRESS_ID INTEGER NOT NULL,
	constraint CUSTOMER_EMAIL_ID_PK primary key ( EMAIL_ID )
);

CREATE TABLE CUSTOMER_CART (
	CART_ID INT auto_increment,
	PRODUCT_ID INT NOT NULL,
	QUANTITY SMALLINT,
    CUSTOMER_EMAIL_ID VARCHAR(50),
	constraint CART_ID_PK primary key (CART_ID)
);

CREATE TABLE STORE(
	STORE_ID INTEGER NOT NULL AUTO_INCREMENT,
	STORE_NAME VARCHAR(50),
	ADDRESS_ID INTEGER NOT NULL,
	constraint STORE_PK primary key ( STORE_Id)
);

CREATE TABLE GEO_LOCATION(
	LOC_ID Int auto_increment,
	LATITUDE FLOAT,
	LONGITUDE FLOAT,
	constraint LOC_ID_PK primary key (LOC_ID)
	);
	
CREATE TABLE PRODUCT(
	PRODUCT_ID INT auto_increment,
	NAME VARCHAR(500) NOT NULL,
	MRP BIGINT NOT NULL,
	DISCOUNT_PERCENT INTEGER default 0,
	AVAILABLE_QUANTITY BIGINT default 0,
	DISCOUNTED_SELLING_PRICE DECIMAL(5,2) NOT NULL,
	WEIGHTS_IN_GMS FLOAT,
	OUT_OF_STOCK BOOLEAN default FALSE,
	QUANTITY SMALLINT NOT NULL,
	constraint PRODUCT_ID_PK primary key ( PRODUCT_ID )
);

CREATE TABLE ORDERS (
	ORDER_ID INT auto_increment,
	ORDER_NUMBER BIGINT NOT NULL,
	DATE_OF_ORDER DATETIME NOT NULL,
	PRODUCT_ID INT NOT NULL,
	QUANTITY INT NOT NULL,
	TOTAL_PRICE FLOAT NOT NULL,
	ADDRESS_ID INT NOT NULL,
	ORDER_STATUS VARCHAR(20) NOT NULL,
	PAYMENT_THROUGH VARCHAR(20) NOT NULL, 
	DATE_OF_DELIVERY DATETIME,
    CUSTOMER_EMAIL_ID VARCHAR(50),
	constraint ORDER_ID_PK primary key (ORDER_ID)
);
